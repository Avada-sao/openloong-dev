14 // SVD based pseudo-inverse
15 Eigen::MatrixXd pseudoInv_SVD(const Eigen::MatrixXd &mat)
16 {
17  typedef typename Eigen::MatrixXd::Scalar Scalar;
18  auto svd = mat.jacobiSvd(Eigen::ComputeFullU | Eigen::ComputeFullV);
19  const auto &singularValues = svd.singularValues();
20  Eigen::MatrixXd singularValuesInv;
21  double maxSgVal{0};
22  for (unsigned int i = 0; i < singularValues.size(); ++i) {
23  if (singularValues(i) > maxSgVal)
24  maxSgVal=singularValues(i);
25  }
26  double tolerance= std::numeric_limits<double>::epsilon() * std::max(mat.rows(), mat.cols()) * maxSgVal;
27  singularValuesInv=Eigen::MatrixXd::Zero(mat.cols(),mat.rows());
28  for (unsigned int i = 0; i < singularValues.size(); ++i) {
29  if (singularValues(i) > tolerance) {
30  singularValuesInv(i, i) = Scalar{1} / singularValues(i);
31  } else {
32  singularValuesInv(i, i) = Scalar{0};
33  }
34  }
35  return svd.matrixV() * singularValuesInv * svd.matrixU().adjoint();
36 };

84 Eigen::Matrix<double, 3, 3> eul2Rot(double roll, double pitch, double yaw) {
85  Eigen::Matrix<double, 3, 3> Rx, Ry, Rz;
86  Rz << cos(yaw), -sin(yaw), 0,
87  sin(yaw), cos(yaw), 0,
88  0, 0, 1;
89  Ry << cos(pitch), 0, sin(pitch),
90  0, 1, 0,
91  -sin(pitch), 0, cos(pitch);
92  Rx << 1, 0, 0,
93  0, cos(roll), -sin(roll),
94  0, sin(roll), cos(roll);
95  return Rz * Ry * Rx;
96 }

120 Eigen::Matrix<double, 3, 1> diffRot(const Eigen::Matrix3d &Rcur, Eigen::Matrix3d &Rdes) {
121  Eigen::Matrix3d R = Rcur.transpose() * Rdes;
122  Eigen::Vector3d w;
123 
124  if (R.isDiagonal(1e-5) && fabs(R(0, 0)) + fabs(R(1, 1)) + fabs(R(2, 2)) - 3 < 1e-3) {
125  w.setZero();
126  } else if (R.isDiagonal(1e-5)) {
127  w << R(0, 0) + 1, R(1, 1) + 1, R(2, 2) + 1;
128  w = w * 3.1415 / 2.0;
129  } else {
130  Eigen::Vector3d l;
131  l << R(2, 1) - R(1, 2), R(0, 2) - R(2, 0), R(1, 0) - R(0, 1);
132  double sita = atan2(l.norm(), R(0, 0) + R(1, 1) + R(2, 2) - 1);
133  w = sita * l / l.norm();
134  }
135  w = Rcur * w;
136  return w;
137 }

139 Eigen::Matrix<double, 4, 1> quat2axisAngle(const Eigen::Quaternion<double> &quat) {
140  // quat must be normolized
141  Eigen::Matrix<double, 4, 1> res;
142  double angle = 2.0 * acos(quat.w());
143  double s = sqrt(1 - quat.w() * quat.w());
144  if (s < 1e-8) {
145  res(0) = quat.x();
146  res(1) = quat.y();
147 // res(2)=quat.z();
148  res(2) = 1; // no rotation now but the axis must not be [0,0,0]
149  } else {
150  res(0) = quat.x() / s;
151  res(1) = quat.y() / s;
152  res(2) = quat.z() / s;
153  }
154  res(3) = angle;
155  return res;
156 }

158 Eigen::Quaterniond intQuat(const Eigen::Quaterniond &quat, const Eigen::Matrix<double, 3, 1> &w) {
159  Eigen::Matrix3d Rcur = quat.normalized().toRotationMatrix();
160  Eigen::Matrix3d Rinc = Eigen::Matrix3d::Identity();
161  double theta = w.norm();
162  if (theta > 1e-4) {
163  Eigen::Vector3d w_norm;
164  w_norm = w / theta;
165  Eigen::Matrix3d a;
166  a << 0, -w_norm(2), w_norm(1),
167  w_norm(0), 0, -w_norm(0),
168  -w_norm(1), w_norm(0), 0;
169  Rinc = Eigen::Matrix3d::Identity() + a * sin(theta) + a * a * (1 - cos(theta));
170  }
171  Eigen::Matrix3d Rend = Rcur * Rinc;
172  Eigen::Quaterniond quatRes;
173  quatRes = Rend;
174  return quatRes;
175 }
