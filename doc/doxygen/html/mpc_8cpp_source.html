88 void MPC::dataBusRead(DataBus &Data) {
89  if (EN) {
90  //set Xd
91  for (int i = 0; i < (mpc_N - 1); i++)
92  Xd.block<nx, 1>(nx * i, 0) = Xd.block<nx, 1>(nx * (i + 1), 0);
93  for (int j = 0; j < 3; j++)
94  Xd(nx * (mpc_N - 1) + j) = Data.js_eul_des(j);
95  for (int j = 0; j < 3; j++)
96  Xd(nx * (mpc_N - 1) + 3 + j) = Data.js_pos_des(j);
97  for (int j = 0; j < 3; j++)
98  Xd(nx * (mpc_N - 1) + 6 + j) = Data.js_omega_des(j);
99  for (int j = 0; j < 3; j++)
100  Xd(nx * (mpc_N - 1) + 9 + j) = Data.js_vel_des(j);
101  }
102  else{
103  for (int j = 0; j < 3; j++)
104  Data.js_eul_des(j) = Data.base_rpy(j);
105  for (int j = 0; j < 3; j++)
106  Data.js_pos_des(j) = Data.base_pos(j);
107  for (int j = 0; j < 3; j++)
108  Data.js_omega_des(j) = Data.base_omega_W(j);
109  for (int j = 0; j < 3; j++)
110  Data.js_vel_des(j) = Data.dq(j);
111 
112  for (int i = 0; i < mpc_N; i++){
113  for (int j = 0; j < 3; j++)
114  Xd(nx * i + j) = Data.js_eul_des(j);
115  for (int j = 0; j < 3; j++)
116  Xd(nx * i + 3 + j) = Data.js_pos_des(j);
117  for (int j = 0; j < 3; j++)
118  Xd(nx * i + 6 + j) = Data.js_omega_des(j);
119  for (int j = 0; j < 3; j++)
120  Xd(nx * i + 9 + j) = Data.js_vel_des(j);
121  }
122  }
123 
124  //set value
125  X_cur.block<3,1>(0,0) = Data.base_rpy;
126  X_cur.block<3,1>(3,0) = Data.q.block<3,1>(0,0);
127  X_cur.block<3,1>(6,0) = Data.dq.block<3,1>(3,0);
128  X_cur.block<3,1>(9,0) = Data.dq.block<3,1>(0,0);
129 
130  R_cur = Data.base_rot;
131 
132  pCoM = X_cur.block<3,1>(3,0);
133  pe.block<3,1>(0,0) = Data.fe_l_pos_W;
134  pe.block<3,1>(3,0) = Data.fe_r_pos_W;
135 
136  pf2com.block<3,1>(0,0) = pe.block<3,1>(0,0) - pCoM;
137  pf2com.block<3,1>(3,0) = pe.block<3,1>(3,0) - pCoM;
138  pf2comd.block<3,1>(0,0) = pe.block<3,1>(0,0) - Xd.block<3,1>(3,0);
139  pf2comd.block<3,1>(3,0) = pe.block<3,1>(3,0) - Xd.block<3,1>(3,0);
140 
141  Ic = Data.inertia;
142 
143  legStateCur = Data.legState;
144  legStateNext = Data.legStateNext;
145  for (int i = 0; i < mpc_N; i++){
146  double aa;
147  aa = i*dt/0.4;
148  double phip;
149  phip = Data.phi + aa;
150  if (phip > 1)
151  legState[i] = legStateNext;
152  else
153  legState[i] = legStateCur;
154  }
155 
156  Eigen::Matrix<double, 3, 3> R_slop;
157  R_slop = eul2Rot(Data.slop(0), Data.slop(1), Data.slop(2));
158  if (legStateCur == DataBus::RSt)
159  R_f2w = Data.fe_r_rot_W;
160  else if (legStateCur == DataBus::LSt)
161  R_f2w = Data.fe_l_rot_W;
162  else
163  R_f2w = R_slop;
164  R_w2f = R_f2w.transpose();
165 }

167 void MPC::cal() {
168  if (EN) {
169  //qp pre
170  Eigen::Matrix<double, 3, 3> eye3m;
171  Eigen::Matrix<double, 3, 1> zero3v;
172  eye3m << 1.0, 0.0, 0.0,
173  0.0, 1.0, 0.0,
174  0.0, 0.0, 1.0;
175  zero3v.setZero();
176  Eigen::Matrix<double, nx, nx> eye12m;
177  eye12m.setZero();
178  for (int i = 0; i < nx; i++)
179  eye12m(i, i) = 1.0;
180 
181  for (int i = 0; i < mpc_N; i++) {
182  R_curz[i] = Rz3(X_cur(2));
183  }
184  for (int i = 0; i < mpc_N; i++) {
185  Ac[i].block<3, 3>(0, 6) = R_curz[i].transpose();
186  Ac[i].block<3, 3>(3, 9) = eye3m;
187  A[i] = eye12m + dt * Ac[i];
188  }
189  for (int i = 0; i < mpc_N; i++) {
190  pf2comi[i] = pf2com;
191  Eigen::Matrix3d Ic_W_inv;
192  Ic_W_inv = (R_curz[i]*Ic* R_curz[i].transpose()).inverse();
193  Bc[i].block<3, 3>(6, 0) = Ic_W_inv*CrossProduct_A(pf2comi[i].block<3, 1>(0, 0));
194  Bc[i].block<3, 3>(6, 3) = Ic_W_inv;
195  Bc[i].block<3, 3>(6, 6) = Ic_W_inv*CrossProduct_A(pf2comi[i].block<3, 1>(3, 0));
196  Bc[i].block<3, 3>(6, 9) = Ic_W_inv;
197  Bc[i].block<3, 3>(9, 0) = eye3m / m;
198  Bc[i].block<3, 3>(9, 6) = eye3m / m;
199  Bc[i]((nx - 1), (nu - 1)) = 1.0 / m;
200  B[i] = dt * Bc[i];
201  }
202  for (int i = 0; i < mpc_N; i++)
203  Aqp.block<nx, nx>(i * nx, 0) = eye12m;
204  for (int i = 0; i < mpc_N; i++)
205  for (int j = 0; j < i + 1; j++)
206  Aqp.block<nx, nx>(i * nx, 0) = A[j] * Aqp.block<nx, nx>(i * nx, 0);
207 
208  for (int i = 0; i < mpc_N; i++)
209  for (int j = 0; j < i + 1; j++)
210  Aqp1.block<nx, nx>(i * nx, j * nx) = eye12m;
211  for (int i = 1; i < mpc_N; i++)
212  for (int j = 0; j < i; j++)
213  for (int k = j + 1; k < (i + 1); k++)
214  Aqp1.block<nx, nx>(i * nx, j * nx) = A[k] * Aqp1.block<nx, nx>(i * nx, j * nx);
215 
216  for (int i = 0; i < mpc_N; i++)
217  Bqp1.block<nx, nu>(i * nx, i * nu) = B[i];
218  Eigen::Matrix<double, nu * mpc_N, nu * ch> Bqp11;
219  Bqp11.setZero();
220  Bqp11.block<nu * ch, nu * ch>(0, 0) = Eigen::MatrixXd::Identity(nu * ch, nu * ch);
221  for (int i = 0; i < (mpc_N - ch); i++)
222  Bqp11.block<nu, nu>(nu * ch + i * nu, nu * (ch - 1)) = Eigen::MatrixXd::Identity(nu, nu);
223 
224  Eigen::Matrix<double, nx * mpc_N, nu * ch> B_tmp;
225  B_tmp = Bqp1 * Bqp11;
226  Bqp = Aqp1 * B_tmp;
227 
228  H = 2 * (Bqp.transpose() * L * Bqp + alpha * K);
229  c = 2 * Bqp.transpose() * L * (Aqp * X_cur - Xd);
230 
231  //friction constraint
232  Eigen::Matrix<double, ncfr_single, 3> Asfr111, Asfr11;
233  Eigen::Matrix<double, ncfr, nu> Asfr1;
234  Eigen::Matrix<double, ncfr * ch, nu * ch> Asfr;
235  Asfr111.setZero();
236  Asfr1.setZero();
237  Asfr.setZero();
238  Asfr111 <<
239  -1.0, 0.0, -1.0 / sqrt(2.0) * miu,
240  1.0, 0.0, -1.0 / sqrt(2.0) * miu,
241  0.0, -1.0, -1.0 / sqrt(2.0) * miu,
242  0.0, 1.0, -1.0 / sqrt(2.0) * miu;
243  Asfr11 = Asfr111 * R_w2f;
244  Asfr1.block<ncfr_single, 3>(0, 0) = Asfr11;
245  Asfr1.block<ncfr_single, 3>(ncfr_single, 6) = Asfr11;
246 
247  for (int i = 0; i < ch; i++)
248  Asfr.block<ncfr, nu>(ncfr * i, i * nu) = Asfr1;
249 
250  //moment constraint x y
251  double sign_xy[4]{1.0, -1.0, -1.0, 1.0};
252  Eigen::Matrix<double, 3, 1> gxyz[4];
253  gxyz[0] << 0.0, 1.0, 0.0;
254  gxyz[1] << 0.0, 1.0, 0.0;
255  gxyz[2] << 1.0, 0.0, 0.0;
256  gxyz[3] << 1.0, 0.0, 0.0;
257  Eigen::Matrix<double, 3, 1> r[4];
258  Eigen::Matrix<double, 3, 1> p[4];
259  Eigen::Matrix<double, ncstxya, 6> Astxy_r[4];
260  Eigen::Matrix<double, ncstxy_single, 6> Astxy11;
261  Eigen::Matrix<double, ncstxy, nu> Astxy1;
262  Eigen::Matrix<double, ncstxy * ch, nu * ch> Astxy;
263  Astxy_r[0].setZero();
264  Astxy_r[1].setZero();
265  Astxy_r[2].setZero();
266  Astxy_r[3].setZero();
267  Astxy11.setZero();
268  Astxy1.setZero();
269  Astxy.setZero();
270 
271  r[0] << 0.0, 1.0, 0.0;
272  r[1] << 0.0, 1.0, 0.0;
273  r[2] << 1.0, 0.0, 0.0;
274  r[3] << 1.0, 0.0, 0.0;
275 
276  p[0] << delta_foot[0], 0.0, 0.0;
277  p[1] << -delta_foot[1], 0.0, 0.0;
278  p[2] << 0.0, delta_foot[2], 0.0;
279  p[3] << 0.0, -delta_foot[3], 0.0;
280 
281  for (int i = 0; i < 4; i++) {
282  Astxy_r[i].block<1, 3>(0, 0) =
283  sign_xy[i] * gxyz[i].transpose() * R_w2f * R_f2w * r[i] * (R_f2w * r[i]).transpose() *
284  CrossProduct_A(R_f2w * p[i]);
285  Astxy_r[i].block<1, 3>(0, 3) = sign_xy[i] * gxyz[i].transpose() * R_w2f;
286  Astxy11.block<ncstxya, 6>(i * ncstxya, 0) = Astxy_r[i];
287  }
288  Astxy1.block<ncstxy_single, 6>(0, 0) = Astxy11;
289  Astxy1.block<ncstxy_single, 6>(ncstxy_single, 6) = Astxy11;
290  for (int i = 0; i < ch; i++)
291  Astxy.block<ncstxy, nu>(ncstxy * i, nu * i) = Astxy1;
292 
293  //moment constraint z
294  Eigen::Matrix<double, ncstza, 6> Astz_r[4];
295  Eigen::Matrix<double, ncstz_single, 6> Astz11;
296  Eigen::Matrix<double, ncstz, nu> Astz1;
297  Eigen::Matrix<double, ncstz * ch, nu * ch> Astz;
298  Astz_r[0].setZero();
299  Astz_r[1].setZero();
300  Astz_r[2].setZero();
301  Astz_r[3].setZero();
302  Astz11.setZero();
303  Astz1.setZero();
304  Astz.setZero();
305 
306  for (int i = 0; i < 4; i++) {
307  Astz_r[i].block<1, 3>(0, 0) = -sqrt(p[i](0) * p[i](0) + p[i](1) * p[i](1) + p[i](2) * p[i](2)) * miu *
308  Eigen::Matrix<double, 1, 3>(0.0, 0.0, 1.0) * R_w2f;
309  Astz_r[i].block<1, 3>(0, 3) = Eigen::Matrix<double, 1, 3>(0.0, 0.0, 1.0) * R_w2f;
310  Astz_r[i].block<1, 3>(1, 0) = Astz_r[i].block<1, 3>(0, 0);
311  Astz_r[i].block<1, 3>(1, 3) = -1*Astz_r[i].block<1, 3>(0, 3);
312  Astz11.block<ncstza, 6>(i * ncstza, 0) = Astz_r[i];
313  }
314  Astz1.block<ncstz_single, 6>(0, 0) = Astz11;
315  Astz1.block<ncstz_single, 6>(ncstz_single, 6) = Astz11;
316  for (int i = 0; i < ch; i++)
317  Astz.block<ncstz, nu>(ncstz * i, nu * i) = Astz1;
318 
319  As.block<ncfr * ch, nu * ch>(0, 0) = Asfr;
320  As.block<ncstxy * ch, nu * ch>(ncfr * ch, 0) = Astxy;
321  As.block<ncstz * ch, nu * ch>(ncfr * ch + ncstxy * ch, 0) = Astz;
322 
323  bs.setZero();
324 
325  //qp
326  Eigen::Matrix<double, nu * ch, 1> Guess_value;
327  Guess_value.setZero();
328  for (int i = 0; i < ch; i++){
329  if (legState[i] == DataBus::DSt) {
330  Guess_value(i * nu + 2) = -0.5 * m * g;
331  Guess_value(i * nu + 8) = -0.5 * m * g;
332  Guess_value(i * nu + 12) = m * g;
333  for (int j = 0; j < 6; j++) {
334  u_low(i * nu + j) = min[j];
335  u_low(i * nu + j + 6) = min[j];
336  u_up(i * nu + j) = max[j];
337  u_up(i * nu + j + 6) = max[j];
338  }
339  u_low(i * nu + 12) = m * g;
340  u_up(i * nu + 12) = m * g;
341  } else if (legState[i] == DataBus::LSt) {
342  Guess_value(i * nu + 2) = -m * g;
343  Guess_value(i * nu + 12) = m * g;
344  for (int j = 0; j < 6; j++) {
345  u_low(i * nu + j) = min[j];
346  u_low(i * nu + j + nu / 2) = 0.0;
347  u_up(i * nu + j) = max[j];
348  u_up(i * nu + j + nu / 2) = 0.0;
349  }
350  u_low(i * nu + 8) = 0.0;
351  u_up(i * nu + 8) = 0.0;
352 
353  u_low(i * nu + 12) = m * g;
354  u_up(i * nu + 12) = m * g;
355  } else if (legState[i] == DataBus::RSt) {
356  Guess_value(i * nu + 8) = -m * g;
357  Guess_value(i * nu + 12) = m * g;
358  for (int j = 0; j < 6; j++) {
359  u_low(i * nu + j) = 0.0;
360  u_low(i * nu + j + nu / 2) = min[j];
361  u_up(i * nu + j) = 0.0;
362  u_up(i * nu + j + nu / 2) = max[j];
363  }
364  u_low(i * nu + 2) = 0.0;
365  u_up(i * nu + 2) = 0.0;
366 
367  u_low(i * nu + 12) = m * g;
368  u_up(i * nu + 12) = m * g;
369  }
370  }
371 
372  qpOASES::returnValue res;
373  nWSR = 1000000;
374  cpu_time = dt;
375 
376  Eigen::Matrix<double, nc * ch, 1> lbA, ubA, one_ch_1;
377  one_ch_1.setOnes();
378  lbA = -1e7 * one_ch_1;
379  ubA = 1e7 * one_ch_1;
380 
381  for (int i = 0; i < ch; i++){
382  if(legState[i] == DataBus::DSt){
383  ubA.block<ncfr, 1>(ncfr * i, 0).setZero();
384  ubA.block<ncstxy, 1>(ncfr * ch + ncstxy * i, 0).setZero();
385  ubA.block<ncstz, 1>(ncfr * ch + ncstxy * ch + ncstz * i, 0).setZero();
386  }
387  else if (legState[i] == DataBus::LSt) {
388  ubA.block<ncfr_single, 1>(ncfr * i, 0).setZero();
389  ubA.block<ncstxy_single, 1>(ncfr * ch + ncstxy * i, 0).setZero();
390  ubA.block<ncstz_single, 1>(ncfr * ch + ncstxy * ch + ncstz * i, 0).setZero();
391  }
392  else if (legState[i] == DataBus::RSt) {
393  ubA.block<ncfr_single, 1>(ncfr * i + ncfr_single, 0).setZero();
394  ubA.block<ncstxy_single, 1>(ncfr * ch + ncstxy * i + ncstxy_single, 0).setZero();
395  ubA.block<ncstz_single, 1>(ncfr * ch + ncstxy * ch + ncstz * i + ncstz_single, 0).setZero();
396  }
397  }
398 
399  copy_Eigen_to_real_t(qp_H, H, nu * ch, nu * ch);
400  copy_Eigen_to_real_t(qp_c, c, 1, nu * ch);
401  copy_Eigen_to_real_t(qp_As, As, nc * ch, nu * ch);
402  copy_Eigen_to_real_t(qp_lbA, lbA, nc * ch, 1);
403  copy_Eigen_to_real_t(qp_ubA, ubA, nc * ch, 1);
404  copy_Eigen_to_real_t(qp_lu, u_low, nu * ch, 1);
405  copy_Eigen_to_real_t(qp_uu, u_up, nu * ch, 1);
406  copy_Eigen_to_real_t(xOpt_iniGuess, Guess_value, nu * ch, 1);
407  res = QP.init(qp_H, qp_c, qp_As, qp_lu, qp_uu, qp_lbA, qp_ubA, nWSR, &cpu_time, xOpt_iniGuess);
408 
409  qpStatus = qpOASES::getSimpleStatus(res);
410  last_nWSR = nWSR;
411  last_cpuTime = cpu_time;
412 
413  qpOASES::real_t xOpt[nu * ch];
414  QP.getPrimalSolution(xOpt);
415  if (qpStatus == 0) {
416  for (int i = 0; i < nu * ch; i++)
417  Ufe(i) = xOpt[i];
418  }
419 
420  dX_cal = Ac[0] * X_cur + Bc[0] * Ufe.block<nu,1>(0,0);
421  Eigen::Matrix<double, nx, 1> delta_X;
422  delta_X.setZero();
423  for (int i = 0; i < 3; i++){
424  delta_X(i) = 0.5*dX_cal(i+6)*dt*dt;
425  delta_X(i+3) = 0.5*dX_cal(i+9)*dt*dt;
426  delta_X(i+6) = dX_cal(i+6)*dt;
427  delta_X(i+9) = dX_cal(i+9)*dt;
428  }
429 
430  X_cal = (Aqp * X_cur + Bqp * Ufe).block<nx,1>(nx*0,0) + delta_X;
431 
432  Ufe_pre = Ufe.block<nu, 1>(0, 0);
433  QP.reset();
434  }

437 void MPC::dataBusWrite(DataBus &Data) {
438  Data.Xd = Xd;
439  Data.X_cur = X_cur;
440  Data.fe_react_tau_cmd = Ufe;
441  Data.X_cal = X_cal;
442  Data.dX_cal = dX_cal;
443 
444  Data.qp_nWSR_MPC = nWSR;
445  Data.qp_cpuTime_MPC = cpu_time;
446  Data.qpStatus_MPC = qpStatus;
447 
448  Data.Fr_ff = Ufe.block<12, 1>(0, 0);
449 
450  double k = 5;
451  Data.des_ddq.block<2, 1>(0, 0) << dX_cal(9), dX_cal(10);
452 
453  Data.des_ddq(5) = k * (Xd(6 + 2) - Data.dq(5));
454 
455  Data.des_dq.block<3, 1>(0, 0) << Xd(9 + 0), Xd(9 + 1), Xd(9 + 2);
456  Data.des_dq.block<2, 1>(3, 0) << 0.0, 0.0;
