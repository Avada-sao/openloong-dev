35 void PriorityTasks::buildPriority(const std::vector<std::string> &taskOrder) {
36  startId= getId(taskOrder[0]);
37  for (int i=0;i<taskOrder.size();i++)
38  {
39  int idCur= getId(taskOrder[i]);
40  if (i==0)
41  taskLib[idCur].parentId=-1;
42  else
43  {
44  int idBef= getId(taskOrder[i-1]);
45  taskLib[idCur].parentId=idBef;
46  }
47  if (i==taskOrder.size()-1)
48  taskLib[idCur].childId=-1;
49  else
50  {
51  int idNxt= getId(taskOrder[i+1]);
52  taskLib[idCur].childId=idNxt;
53  }

66 void PriorityTasks::computeAll(const Eigen::VectorXd &des_delta_q,const Eigen::VectorXd &des_dq, const Eigen::VectorXd &des_ddq, const Eigen::MatrixXd &dyn_M, const Eigen::MatrixXd &dyn_M_inv, const Eigen::VectorXd &dq) {
67  int curId=startId;
68  int parentId=taskLib[curId].parentId;
69  int childId=taskLib[curId].childId;
70  for (int i=0;i<taskLib.size();i++)
71  {
72  if (parentId==-1){
73  taskLib[curId].N=Eigen::MatrixXd::Identity(taskLib[curId].J.cols(),taskLib[curId].J.cols());
74  taskLib[curId].Jpre=taskLib[curId].J*taskLib[curId].N;
75  taskLib[curId].delta_q=des_delta_q+ pseudoInv_right_weighted(taskLib[curId].Jpre,taskLib[curId].W)*taskLib[curId].errX;
76  taskLib[curId].dq=des_dq;
77  Eigen::VectorXd ddxcmd= taskLib[curId].ddxDes + taskLib[curId].kp * taskLib[curId].errX+taskLib[curId].kd*taskLib[curId].derrX;
78  taskLib[curId].ddq= des_ddq + dyn_pseudoInv(taskLib[curId].Jpre,dyn_M_inv,true) * (ddxcmd - taskLib[curId].J * dq);
79 // std::cout<<taskLib[curId].taskName<<std::endl<<taskLib[curId].delta_q.transpose()<<std::endl;
80  }
81  else{
82  taskLib[curId].N=taskLib[parentId].N*
83  (Eigen::MatrixXd::Identity(taskLib[parentId].Jpre.cols(),taskLib[parentId].Jpre.cols())- pseudoInv_right_weighted(taskLib[parentId].Jpre,taskLib[parentId].W)*taskLib[parentId].Jpre);
84  taskLib[curId].Jpre=taskLib[curId].J*taskLib[curId].N;
85  taskLib[curId].delta_q=taskLib[parentId].delta_q+ pseudoInv_right_weighted(taskLib[curId].Jpre,taskLib[curId].W)*(taskLib[curId].errX-
86  taskLib[curId].J*taskLib[parentId].delta_q);
87  taskLib[curId].dq=taskLib[parentId].dq+ pseudoInv_right_weighted(taskLib[curId].Jpre,taskLib[curId].W)*(taskLib[curId].dxDes-
88  taskLib[curId].J*taskLib[parentId].dq);
89  Eigen::VectorXd ddxcmd= taskLib[curId].ddxDes + taskLib[curId].kp * taskLib[curId].errX+taskLib[curId].kd*taskLib[curId].derrX;
90  taskLib[curId].ddq= taskLib[parentId].ddq + dyn_pseudoInv(taskLib[curId].Jpre,dyn_M_inv,true) *
91  (ddxcmd-taskLib[curId].dJ*dq-taskLib[curId].J*taskLib[parentId].ddq);
92 // std::cout<<taskLib[curId].taskName<<std::endl<<taskLib[curId].delta_q.transpose()<<std::endl;
93  }
94 // printf("task: %s\n", taskLib[curId].taskName.c_str());
95 // Eigen::FullPivLU<Eigen::MatrixXd> lu_decomp(taskLib[curId].Jpre);
96 // printf("taskJacobian rank: %d, rows: %d\n", lu_decomp.rank(), taskLib[curId].Jpre.rows());
97  if (childId!=-1){
98  parentId=curId;
99  curId=childId;
100  childId=taskLib[curId].childId;
101  }
102  else
103  break;
104  }
105  out_delta_q=taskLib[curId].delta_q;
106  out_dq=taskLib[curId].dq;
107  out_ddq=taskLib[curId].ddq;
108 }
