52 void MJ_Interface::updateSensorValues() {
53  for (int i=0;i<jointNum;i++){
54  motor_pos_Old[i]=motor_pos[i];
55  motor_pos[i]=mj_data->qpos[jntId_qpos[i]];
56  motor_vel[i]=mj_data->qvel[jntId_qvel[i]];
57  }
58  for (int i=0;i<4;i++)
59  baseQuat[i]=mj_data->sensordata[orientataionSensorId+i];
60  double tmp=baseQuat[0];
61  baseQuat[0]=baseQuat[1];
62  baseQuat[1]=baseQuat[2];
63  baseQuat[2]=baseQuat[3];
64  baseQuat[3]=tmp;
65 
66  rpy[0]= atan2(2*(baseQuat[3]*baseQuat[0]+baseQuat[1]*baseQuat[2]),1-2*(baseQuat[0]*baseQuat[0]+baseQuat[1]*baseQuat[1]));
67  rpy[1]= asin(2*(baseQuat[3]*baseQuat[1]-baseQuat[0]*baseQuat[2]));
68  rpy[2]= atan2(2*(baseQuat[3]*baseQuat[2]+baseQuat[0]*baseQuat[1]),1-2*(baseQuat[1]*baseQuat[1]+baseQuat[2]*baseQuat[2]));
69 
70  for (int i=0;i<3;i++)
71  {
72  double posOld=basePos[i];
73  basePos[i]=mj_data->xpos[3*baseBodyId+i];
74  baseAcc[i]=mj_data->sensordata[accSensorId+i];
75  baseAngVel[i]=mj_data->sensordata[gyroSensorId+i];
76  baseLinVel[i]=(basePos[i]-posOld)/(mj_model->opt.timestep);
77  }
78 }
