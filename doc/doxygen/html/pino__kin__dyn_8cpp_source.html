80 void Pin_KinDyn::dataBusRead(const DataBus &robotState) {
81  // For Pinocchio: The base translation part is expressed in the parent frame (here the world coordinate system)
82  // while its velocity is expressed in the body coordinate system.
83  // https://github.com/stack-of-tasks/pinocchio/issues/1137
84  // q = [global_base_position, global_base_quaternion, joint_positions]
85  // v = [local_base_velocity_linear, local_base_velocity_angular, joint_velocities]
86  q=robotState.q;
87  dq=robotState.dq;
88  dq.block(0,0,3,1)=robotState.base_rot.transpose()*dq.block(0,0,3,1);
89  dq.block(3,0,3,1)=robotState.base_rot.transpose()*dq.block(3,0,3,1);
90  ddq=robotState.ddq;
91 }

218 Eigen::Quaterniond Pin_KinDyn::intQuat(const Eigen::Quaterniond &quat, const Eigen::Matrix<double, 3, 1> &w) {
219  Eigen::Matrix3d Rcur=quat.normalized().toRotationMatrix();
220  Eigen::Matrix3d Rinc=Eigen::Matrix3d::Identity();
221  double theta=w.norm();
222  if (theta>1e-8) {
223  Eigen::Vector3d w_norm;
224  w_norm = w / theta;
225  Eigen::Matrix3d a;
226  a << 0, -w_norm(2), w_norm(1),
227  w_norm(0), 0, -w_norm(0),
228  -w_norm(1), w_norm(0), 0;
229  Rinc=Eigen::Matrix3d::Identity()+a*sin(theta)+a*a*(1-cos(theta));
230  }
231  Eigen::Matrix3d Rend=Rcur*Rinc;
232  Eigen::Quaterniond quatRes;
233  quatRes=Rend;
234  return quatRes;
235 }

237 // intergrate the q with dq, for floating base dynamics
238 Eigen::VectorXd Pin_KinDyn::integrateDIY(const Eigen::VectorXd &qI, const Eigen::VectorXd &dqI) {
239  Eigen::VectorXd qRes=Eigen::VectorXd::Zero(model_nv+1);
240  Eigen::Vector3d wDes;
241  wDes<<dqI(3),dqI(4),dqI(5);
242  Eigen::Quaterniond quatNew,quatNow;
243  quatNow.x()=qI(3);
244  quatNow.y()=qI(4);
245  quatNow.z()=qI(5);
246  quatNow.w()=qI(6);
247  quatNew= intQuat(quatNow,wDes);
248  qRes=qI;
249  qRes(0)+=dqI(0);qRes(1)+=dqI(1);qRes(2)+=dqI(2);
250  qRes(3)=quatNew.x();qRes(4)=quatNew.y();qRes(5)=quatNew.z();qRes(6)=quatNew.w();
251  for (int i=0;i<model_nv-6;i++)
252  qRes(7+i)+=dqI(6+i);
253  return qRes;
254 }

256 // update dynamic parameters, M*ddq+C*dq+G=tau
257 void Pin_KinDyn::computeDyn() {
258  // cal M
259  pinocchio::crba(model_biped, data_biped, q);
260  // Pinocchio only gives half of the M, needs to restore it here
261  data_biped.M.triangularView<Eigen::Lower>() = data_biped.M.transpose().triangularView<Eigen::Lower>();
262  dyn_M = data_biped.M;
263 
264  // cal Minv
265  pinocchio::computeMinverse(model_biped, data_biped, q);
266  data_biped.Minv.triangularView<Eigen::Lower>() = data_biped.Minv.transpose().triangularView<Eigen::Lower>();
267  dyn_M_inv=data_biped.Minv;
268 
269  // cal C
270  pinocchio::computeCoriolisMatrix(model_biped, data_biped, q, dq);
271  dyn_C = data_biped.C;
272 
273  // cal G
274  pinocchio::computeGeneralizedGravity(model_biped, data_biped, q);
275  dyn_G = data_biped.g;
276 
277  // cal Ag, Centroidal Momentum Matrix. First three rows: linear, other three rows: angular
278  pinocchio::dccrba(model_biped,data_biped,q,dq);
279  pinocchio::computeCentroidalMomentum(model_biped,data_biped,q,dq);
280  dyn_Ag=data_biped.Ag;
281  dyn_dAg=data_biped.dAg;
282 
283  // cal nonlinear item
284  dyn_Non=dyn_C*dq+dyn_G;
285 
286  // cal I
287  pinocchio::ccrba(model_biped, data_biped, q, dq);
288  inertia = data_biped.Ig.inertia().matrix();
289 
290  // cal CoM
291  CoM_pos = data_biped.com[0];
292 // std::cout<<"CoM_W"<<std::endl;
293 // std::cout<<CoM_pos.transpose()<<std::endl;
294 
295  Eigen::MatrixXd Mpj, Mpj_inv; // transform into world frame
296  Mpj=Eigen::MatrixXd::Identity(model_nv,model_nv);
297  Mpj_inv=Eigen::MatrixXd::Identity(model_nv,model_nv);
298  Mpj.block(0,0,3,3)=base_rot.transpose();
299  Mpj.block(3,3,3,3)=base_rot.transpose();
300  Mpj_inv.block(0,0,3,3)=base_rot;
301  Mpj_inv.block(3,3,3,3)=base_rot;
302  dyn_M=Mpj_inv*dyn_M*Mpj;
303  dyn_M_inv=Mpj_inv*dyn_M_inv*Mpj;
304  dyn_C=Mpj_inv*dyn_C*Mpj;
305  dyn_G=Mpj_inv*dyn_G;
306  dyn_Non=Mpj_inv*dyn_Non;
307 
308 }

310 // Inverse kinematics for leg posture. Note: the Rdes and Pdes are both w.r.t the baselink coordinate in body frame!
311 Pin_KinDyn::IkRes
312 Pin_KinDyn::computeInK_Leg(const Eigen::Matrix3d &Rdes_L, const Eigen::Vector3d &Pdes_L, const Eigen::Matrix3d &Rdes_R,
313  const Eigen::Vector3d &Pdes_R) {
314  const pinocchio::SE3 oMdesL(Rdes_L, Pdes_L);
315  const pinocchio::SE3 oMdesR(Rdes_R, Pdes_R);
316  // arm-l: 0-6, arm-r: 7-13, head: 14,15 waist: 16-18, leg-l: 19-24, leg-r: 25-30
317  Eigen::VectorXd qIk=Eigen::VectorXd::Zero(model_biped_fixed.nv); // initial guess
318  qIk[22]=-0.1;
319  qIk[28]=-0.1;
320 
321  const double eps = 1e-4;
322  const int IT_MAX = 100;
323  const double DT = 7e-1;
324  const double damp = 5e-3;
325  Eigen::MatrixXd JL(6,model_biped_fixed.nv);
326  Eigen::MatrixXd JR(6,model_biped_fixed.nv);
327  Eigen::MatrixXd JCompact(12,model_biped_fixed.nv);
328  JL.setZero();
329  JR.setZero();
330  JCompact.setZero();
331 
332  bool success = false;
333  Eigen::Matrix<double, 6, 1> errL, errR;
334  Eigen::Matrix<double, 12,1> errCompact;
335  Eigen::VectorXd v(model_biped_fixed.nv);
336 
337  pinocchio::JointIndex J_Idx_l, J_Idx_r;
338  J_Idx_l = l_ankle_joint_fixed;
339  J_Idx_r = r_ankle_joint_fixed;
340  int itr_count{0};
341  for (itr_count=0;; itr_count++)
342  {
343  pinocchio::forwardKinematics(model_biped_fixed,data_biped_fixed,qIk);
344  const pinocchio::SE3 iMdL = data_biped_fixed.oMi[J_Idx_l].actInv(oMdesL);
345  const pinocchio::SE3 iMdR = data_biped_fixed.oMi[J_Idx_r].actInv(oMdesR);
346  errL = pinocchio::log6(iMdL).toVector(); // in joint frame
347  errR = pinocchio::log6(iMdR).toVector(); // in joint frame
348  errCompact.block<6,1>(0,0)=errL;
349  errCompact.block<6,1>(6,0)=errR;
350  if(errCompact.norm() < eps)
351  {
352  success = true;
353  break;
354  }
355  if (itr_count >= IT_MAX)
356  {
357  success = false;
358  break;
359  }
360 
361  pinocchio::computeJointJacobian(model_biped_fixed,data_biped_fixed,qIk,J_Idx_l,JL); // JL in joint frame
362  pinocchio::computeJointJacobian(model_biped_fixed,data_biped_fixed,qIk,J_Idx_r,JR); // JR in joint frame
363  Eigen::MatrixXd W;
364  W=Eigen::MatrixXd::Identity(model_biped_fixed.nv, model_biped_fixed.nv); // weighted matrix
365  // arm-l: 0-6, arm-r: 7-13, head: 14,15 waist: 16-18, leg-l: 19-24, leg-r: 25-30
366 // W(16,16)=0.001; // use a smaller value to make the solver try not to use waist joint
367 // W(17,17)=0.001;
368 // W(18,18)=0.001;
369  JL.block(0,16,6,3).setZero();
370  JR.block(0,16,6,3).setZero();
371  pinocchio::Data::Matrix6 JlogL;
372  pinocchio::Data::Matrix6 JlogR;
373  pinocchio::Jlog6(iMdL.inverse(), JlogL);
374  pinocchio::Jlog6(iMdR.inverse(), JlogR);
375  JL = -JlogL * JL;
376  JR = -JlogR * JR;
377  JCompact.block(0,0,6,model_biped_fixed.nv)=JL;
378  JCompact.block(6,0,6,model_biped_fixed.nv)=JR;
379  // pinocchio::Data::Matrix6 JJt;
380  Eigen::Matrix<double,12,12> JJt;
381  JJt.noalias() = JCompact * W * JCompact.transpose();
382  JJt.diagonal().array() += damp;
383  v.noalias() = - W * JCompact.transpose() * JJt.ldlt().solve(errCompact);
384  qIk = pinocchio::integrate(model_biped_fixed,qIk,v*DT);
385  }
386 
387  IkRes res;
388  res.err=errCompact;
389  res.itr=itr_count;
390 
391  if(success){
392  res.status=0;
393  }
394  else{
395  res.status=-1;
396  }

402 Pin_KinDyn::IkRes
403 Pin_KinDyn::computeInK_Hand(const Eigen::Matrix3d &Rdes_L, const Eigen::Vector3d &Pdes_L, const Eigen::Matrix3d &Rdes_R,
404  const Eigen::Vector3d &Pdes_R) {
405  const pinocchio::SE3 oMdesL(Rdes_L, Pdes_L);
406  const pinocchio::SE3 oMdesR(Rdes_R, Pdes_R);
407  Eigen::VectorXd qIk=Eigen::VectorXd::Zero(model_biped_fixed.nv); // initial guess
408  // arm-l: 0-6, arm-r: 7-13, head: 14,15 waist: 16-18, leg-l: 19-24, leg-r: 25-30
409  qIk.block<7,1>(0,0)<< 0.433153883479341, 1.11739345867607, 1.88491913406236,
410  0.802378252758275, 1.22726400279662, 0.0249797771339966, -0.0875282610654057;
411 
412  qIk.block<7,1>(7,0)<<-0.433152540054138, -1.11739347975224, -1.88492038240761,
413  0.802375980602373, -1.22726323451626, 0.0249795712262396, 0.0875271396314979;
414 
415  const double eps = 1e-4;
416  const int IT_MAX = 100;
417  const double DT = 6e-1;
418  const double damp = 1e-2;
419  Eigen::MatrixXd JL(6,model_biped_fixed.nv);
420  Eigen::MatrixXd JR(6,model_biped_fixed.nv);
421  Eigen::MatrixXd JCompact(12,model_biped_fixed.nv);
422  JL.setZero();
423  JR.setZero();
424  JCompact.setZero();
425 
426  bool success = false;
427  Eigen::Matrix<double, 6, 1> errL, errR;
428  Eigen::Matrix<double, 12,1> errCompact;
429  Eigen::VectorXd v(model_biped_fixed.nv);
430 
431  pinocchio::JointIndex J_Idx_l, J_Idx_r;
432  J_Idx_l = l_hand_joint_fixed;
433  J_Idx_r = r_hand_joint_fixed;
434  int itr_count{0};
435  for (itr_count=0;; itr_count++)
436  {
437  pinocchio::forwardKinematics(model_biped_fixed,data_biped_fixed,qIk);
438  const pinocchio::SE3 iMdL = data_biped_fixed.oMi[J_Idx_l].actInv(oMdesL);
439  const pinocchio::SE3 iMdR = data_biped_fixed.oMi[J_Idx_r].actInv(oMdesR);
440  errL = pinocchio::log6(iMdL).toVector(); // in joint frame
441  errR = pinocchio::log6(iMdR).toVector(); // in joint frame
442  errCompact.block<6,1>(0,0)=errL;
443  errCompact.block<6,1>(6,0)=errR;
444  if(errCompact.norm() < eps)
445  {
446  success = true;
447  break;
448  }
449  if (itr_count >= IT_MAX)
450  {
451  success = false;
452  break;
453  }
454 
455  pinocchio::computeJointJacobian(model_biped_fixed,data_biped_fixed,qIk,J_Idx_l,JL); // JL in joint frame
456  pinocchio::computeJointJacobian(model_biped_fixed,data_biped_fixed,qIk,J_Idx_r,JR); // JR in joint frame
457  pinocchio::Data::Matrix6 JlogL;
458  pinocchio::Data::Matrix6 JlogR;
459  pinocchio::Jlog6(iMdL.inverse(), JlogL);
460  pinocchio::Jlog6(iMdR.inverse(), JlogR);
461  JL = -JlogL * JL;
462  JR = -JlogR * JR;
463  JCompact.block(0,0,6,model_biped_fixed.nv)=JL;
464  JCompact.block(6,0,6,model_biped_fixed.nv)=JR;
465  // pinocchio::Data::Matrix6 JJt;
466  Eigen::Matrix<double,12,12> JJt;
467  JJt.noalias() = JCompact * JCompact.transpose();
468  JJt.diagonal().array() += damp;
469  v.noalias() = - JCompact.transpose() * JJt.ldlt().solve(errCompact);
470  qIk = pinocchio::integrate(model_biped_fixed,qIk,v*DT);
471  }
472 
473  IkRes res;
474  res.err=errCompact;
475  res.itr=itr_count;
476 
477  if(success){
478  res.status=0;
479  }
480  else{
481  res.status=-1;
482  }
483 
484  res.jointPosRes=qIk;
485  return res;
486 }
