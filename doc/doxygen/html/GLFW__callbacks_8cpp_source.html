60  cam.azimuth = arr_view[0];
61  cam.elevation = arr_view[1];
62  cam.distance = arr_view[2];
63  cam.lookat[0] = arr_view[3];
64  cam.lookat[1] = arr_view[4];
65  cam.lookat[2] = arr_view[5];
66

68  cam.lookat[2] += 0.8;
69  cam.type = mjCAMERA_TRACKING;
70  cam.trackbodyid = 1;
71  }
72  else
73  cam.type = mjCAMERA_FREE;
74 
75  //mjv_defaultCamera(&cam);

77  mjv_defaultScene(&scn);

80  mjr_makeContext(mj_model, &con, mjFONTSCALE_150); // model-specific context
81  mjv_moveCamera(mj_model, mjMOUSE_ROTATE_H, 0.0, 0.0, &scn, &cam);
82 
83  // install GLFW mouse and keyboard callbacks
84  glfwSetWindowUserPointer(window, this);
85  glfwSetWindowCloseCallback(window, window_close_callback);
86  glfwSetKeyCallback(window, keyboard);
87  glfwSetCursorPosCallback(window, mouse_move);
88  glfwSetMouseButtonCallback(window, mouse_button);
89  glfwSetScrollCallback(window, scroll);
90 
91  save_video=saveVideo;
92  if (save_video)
93  {
94  image_rgb_ = (unsigned char*)malloc(3*width*height*sizeof(unsigned char));
95  image_depth_ = (float*)malloc(sizeof(float)*width*height);
96 
97  // create output rgb file

144 {
145  // backspace: reset simulation

184 // mouse move callback
185 void UIctr::Mouse_move(double xpos, double ypos)
186 {
187  // no buttons down: nothing to do
188  if( !button_left && !button_middle && !button_right )
189  return;
190 
191  // compute mouse displacement, save
192  double dx = xpos - lastx;
193  double dy = ypos - lasty;
194  lastx = xpos;
195  lasty = ypos;
196 
197  // get current window size
198  int width, height;
199  glfwGetWindowSize(window, &width, &height);
200 
201  // get shift key state
202  bool mod_shift = (glfwGetKey(window, GLFW_KEY_LEFT_SHIFT)==GLFW_PRESS ||
203  glfwGetKey(window, GLFW_KEY_RIGHT_SHIFT)==GLFW_PRESS);
204 
205  // determine action based on mouse button
206  mjtMouse action;
207  if( button_right )
208  action = mod_shift ? mjMOUSE_MOVE_H : mjMOUSE_MOVE_V;
209  else if( button_left )
210  action = mod_shift ? mjMOUSE_ROTATE_H : mjMOUSE_ROTATE_V;
211  else
212  action = mjMOUSE_ZOOM;
213 
214  // move camera
215  mjv_moveCamera(mj_model, action, dx/height, dy/height, &scn, &cam);
216 }
