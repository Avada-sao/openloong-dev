165  // update q according to sensor values, must update sensor values before
166  void updateQ(){
167  base_omega_W << baseAngVel[0],baseAngVel[1],baseAngVel[2];
168  auto Rcur= eul2Rot(rpy[0], rpy[1], rpy[2]);
169  base_omega_W=Rcur*base_omega_W;
170 
171  auto quatNow=eul2quat(rpy[0], rpy[1], rpy[2]);
172  q(0)=basePos[0];
173  q(1)=basePos[1];
174  q(2)=basePos[2];
175  q(3)=quatNow.x();
176  q(4)=quatNow.y();
177  q(5)=quatNow.z();
178  q(6)=quatNow.w();
179  for (int i=0;i<model_nv-6;i++)
180  q(i+7)=motors_pos_cur[i];
181  // For THIS project: The base translation and rotation parts are expressed in the world frame
182  // q = [global_base_position, global_base_quaternion, joint_positions]
183  // v = [global_base_velocity_linear, global_base_velocity_angular, joint_velocities]
184  Eigen::Vector3d vCoM_L;
185  vCoM_L << baseLinVel[0],baseLinVel[1],baseLinVel[2];
186  dq.block<3,1>(0,0)= vCoM_L;
187  dq.block<3,1>(3,0) << base_omega_W[0],base_omega_W[1],base_omega_W[2];
188  for (int i=0;i<model_nv-6;i++)
189  {
190  dq(i+6)=motors_vel_cur[i];
191  }
192 
193  base_pos<<q(0),q(1),q(2);
194  base_rpy << rpy[0], rpy[1], rpy[2];
195  base_rot=Rcur;
196  qOld=q;
197  }

218  Eigen::Matrix<double, 3, 3> eul2Rot(double roll, double pitch, double yaw) {
219  Eigen::Matrix<double,3,3> Rx,Ry,Rz;
220  Rz<<cos(yaw),-sin(yaw),0,
221  sin(yaw),cos(yaw),0,
222  0,0,1;
223  Ry<<cos(pitch),0,sin(pitch),
224  0,1,0,
225  -sin(pitch),0,cos(pitch);
226  Rx<<1,0,0,
227  0,cos(roll),-sin(roll),
228  0,sin(roll),cos(roll);
229  return Rz*Ry*Rx;
230  }

232  Eigen::Quaterniond eul2quat(double roll, double pitch, double yaw) {
233  Eigen::Matrix3d R= eul2Rot(roll,pitch,yaw);
234  Eigen::Quaternion<double> quatCur;
235  quatCur = R; //rotation matrix converted to quaternion
236  Eigen::Quaterniond resQuat;
237  resQuat=quatCur;
238  return resQuat;
239  }
