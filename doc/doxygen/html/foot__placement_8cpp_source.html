31 void FootPlacement::getSwingPos() {
32  Eigen::Matrix3d KP;
33  KP.setZero();
34  KP(0,0)=kp_vx;KP(1,1)=kp_vy;KP(2,2)=0;
35 
36  // for linear velocity
37  posDes_W=hipPos_W+KP*(desV_W-curV_W)*(-1)+0.5*tSwing*curV_W+
38  curV_W*(1-phi)*tSwing;
39 
40  // for angular veloctity
41  double thetaF;
42  thetaF=yawCur+theta0+omegaZ_W*(1-phi)*tSwing+0.5*omegaZ_W*tSwing+kp_wz*(omegaZ_W-desWz_W);
43  posDes_W(0)+=0.5*hip_width* (cos(thetaF)-cos(yawCur+theta0));
44  posDes_W(1)+=0.5*hip_width* (sin(thetaF)-sin(yawCur+theta0));
45 
46  posDes_W(0)=posDes_W(0) - 0.02;
47 // posDes_W(2)=STPos_W(2)-0.04;
48  posDes_W(2)=base_pos(2)-legLength-0.03;
49 
50  double yOff=0.01; // positive for moving the leg inside
51  if (legState==DataBus::LSt)
52  posDes_W(1)+=yOff;
53  else if (legState==DataBus::RSt)
54  posDes_W(1)-=yOff;
55 
56  // cycloid trajectories
57  if (phi < 1.0){
58  pDesCur[0]=posStart_W(0)+(posDes_W(0)-posStart_W(0))/(2*3.1415)*(2*3.1415*phi-sin(2*3.1415*phi));
59  pDesCur[1]=posStart_W(1)+(posDes_W(1)-posStart_W(1))/(2*3.1415)*(2*3.1415*phi-sin(2*3.1415*phi));
60  }
61 // pDesCur[2]=posStart_W(2)+stepHeight*0.5*(1-cos(2*3.1415*phi))+(posDes_W(2)-posStart_W(2))/(2*3.1415)*(2*3.1415*phi-sin(2*3.1415*phi));
62 
63  pDesCur[2]=posStart_W(2)+Trajectory(0.2, stepHeight, posDes_W(2)-posStart_W(2));
64 }

66 double FootPlacement::Trajectory(double phase, double des1, double des2){
67  Bezier_1D Bswpid;
68  double para0 = 5, para1 = 3;
69  for (int i = 0; i < para0; i ++)
70  Bswpid.P.push_back(0.0);
71  for (int i = 0; i < para1; i ++)
72  Bswpid.P.push_back(1.0);
73 
74  double Bsw1, Bsw2;
75  Bsw1 = Bswpid.getOut(phi/phase);
76  Bsw2 = Bswpid.getOut((1.4 - phi)/(1.4 - phase));
77 
78  double output;
79  if (phi < phase)
80  output = des1*Bsw1;
81  else{
82  if (Bsw2 > 0)
83  output = des1 * Bsw2 + des2 * (1.0 - Bsw2);
84  else
85  output = des2;
86  }
87 
88  return output;
89 }
