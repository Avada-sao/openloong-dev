10 DataLogger::DataLogger(std::string fileNameIn) {
11  filePath=fileNameIn;
12  size_t lastSlashPos = filePath.find_last_of('/');
13  fileFolder=filePath.substr(0, lastSlashPos);
14  fileName=filePath.substr(lastSlashPos + 1);
15  file_handler = quill::file_handler(filePath, "w");
16  file_handler->set_pattern(QUILL_STRING("%(message)")); // timestamp's timezone
17  quill::set_default_logger_handler(file_handler);
18  dl = quill::create_logger("logger", file_handler);
19  quill::start();
20 }

22 void DataLogger::addIterm(const std::string &name, const int &len) {
23  auto it = std::find(recItemName.begin(), recItemName.end(), name);
24  if (it != recItemName.end()) {
25  std::cout << name<< " has already been used!!!!!"<< std::endl;
26  throw std::runtime_error("Failed to add rec item.");
27  }
28  recItemName.push_back(name);
29  recItemLen.push_back(len);
30  recItemStartCol.push_back(colCout);
31  recItemEndCol.push_back(colCout+len-1);
32  colCout+=len;
33 }

35 void DataLogger::finishItermAdding() {
36  std::string insFileName=fileFolder+"/matlabReadDataScript.txt";

38 
39  if (outFile.is_open()) {

41  outFile << "dataRec=load('" << fileName << "');" << std::endl;
42  for (size_t i = 0; i < recItemName.size(); ++i) {
43  outFile << recItemName[i] << "=dataRec(:,"<<recItemStartCol[i]+1<<":"<<recItemEndCol[i]+1<<");"<<std::endl;
44  }
45  outFile.close();
46  } else {
47  std::cerr << "unable to open matlabReadDataScript.txt\n";
48  }
49  recValue.resize(colCout,0.0);
50  isItemDataIn.resize(recItemName.size(), false);
51 }

58 void DataLogger::recItermData(const std::string &name, double *dataIn) {
59  auto it = std::find(recItemName.begin(), recItemName.end(), name);
60  if (it == recItemName.end()) {
61  std::cout << name<< " has not been added!!!!!"<< std::endl;
62  throw std::runtime_error("Failed to rec item.");
63  }
64  int curIdx=std::distance(recItemName.begin(), it);
65  for (int i=0;i<recItemLen[curIdx];i++)
66  {
67  recValue[i+recItemStartCol[curIdx]]=dataIn[i];
68  }

70 }
